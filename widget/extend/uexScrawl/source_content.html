appcan.button("#photo", "btn-act", function() {
    var data = {
        min : 1,
        max : 1,
        quality : 0.5,
        detailedInfo : false
    }
    var json = JSON.stringify(data);
    uexImage.openPicker(json);
    uexImage.onPickerClosed = function(info) {
        var img_url = JSON.parse(info).data[0];
        var params = {
            "src" : img_url
        };
        uexScrawl.open(JSON.stringify(params));
    }
    $("#img").addClass('uhide');
})
