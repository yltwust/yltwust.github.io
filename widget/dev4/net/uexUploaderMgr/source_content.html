        var getRadioValue = function(name) {
            var radios = document.getElementsByName(name);
            var value;
            for (var i = 0; i < radios.length; i++) {
                var radio = radios[i];
                if (radio.checked) {
                    value = radio.value;
                }
            }
            return value;

        }
